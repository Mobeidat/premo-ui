## Install

```shell
npm install vue-premo-ui -S
```

## Quick Start

```javascript
import Vue from 'vue'
import VuePremoUi from 'vue-premo-ui'

Vue.use(VueMazelUi)

// or
import {
  PremoSelect,
  PremoButton
  // ...
} from 'vue-mazel-ui'

Vue.component(PremoSelect.name, PremoSelect)
Vue.component(PremoButton.name, PremoButton)
```
